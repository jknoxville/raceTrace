
enum MessageType {
	PayloadMessage = 0;
	RequestMessage = 1;
}

message DatapointSet {
	repeated Datapoint datapoint = 1;
}

message Datapoint {
	required int32 aboutDeviceID = 1;
	required int32 logicalTime = 2;
	required float xCoord = 3;
	required float yCoord = 4;
	required float altitude = 5;
}

message RequestSet {
	repeated Request request = 1;
}

message Request {
	repeated int32 index = 1;
}

message Packet {
	required MessageType type = 1;
	required int32 fromDeviceID = 2;
	optional DatapointSet datapoints = 3
	optional RequestSet requests = 4;
}
